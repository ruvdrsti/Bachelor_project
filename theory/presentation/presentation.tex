    \begin{equation}
        \Delta^{CUHF} = \begin{cases}

            \Delta^{UHF}, & \mbox{anywhere else}
        \end{cases}
    \end{equation}

      \end{equation}     
    \begin{equation}\label{eq:lincomb}
        |\Psi_0\rangle = c_0|\Phi_0\rangle + \sum_{ar}|\Phi_a^r\rangle + \sum_{a<b,r<s}c_{ab}^{rs}|\Phi^{rs}_{ab} \rangle + \cdots

      \end{equation}
\end{frame}
